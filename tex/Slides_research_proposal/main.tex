\begin{frame}[allowframebreaks, noframenumbering]{Appendix}
\begin{block}{Proposition [Comparing discrete and continuous trajectories] ~\cite{barrett,smith}}
For a step size $\alpha\in\R_{>0}$, $t\in \R_{>0}$
\[
\theta_{t+\alpha}=\theta_t - \alpha \nabla\ell(\theta_t)\quad \text{and }\quad \frac{\mathrm{d} \theta}{\de t} = -\nabla\ell(\theta(t))
\]

Up to order $\alpha$, gradient descent corresponds to gradient flow of:
\begin{equation*}
\ell_{discr=contin}(\theta) =\ell(\theta)+\frac{\alpha}{4}\big\lVert\nabla\ell(\theta)\big\rVert^2
\end{equation*}
\end{block}

Proof:

    Consider the usual (discrete-time) update rule:
    \begin{equation*}
        \theta_{\tau+1} = \theta_\tau -\alpha\nabla\ell(\theta_\tau)
    \end{equation*}
for $\tau \in \N$, $\alpha\in\R_{>0}$ step size.
\\
Let $t=\alpha\tau$ and consider the finite-difference
\[
\theta_{t+\alpha} - \theta_t = -\alpha\nabla\ell(\theta_t).
\]
Now, taking the limit we obtain the continuous-time update rule (where we omit the dependence of $\theta$ on $t$ to ease notation)
\begin{align*}
\lim_{\alpha\to0} \frac{\theta_{t+\alpha} - \theta_t}{\alpha} = \frac{\de \theta}{\de t}.
\end{align*}
Then,
\begin{equation}
    \frac{\de \theta}{\de t}= -\nabla\ell(\theta).
\end{equation}
Allowing
\[
\frac{\de \theta}{\de t}= -\nabla\ell(\theta) + \alpha g(\theta),
\]
we want to find the correction term $g$ such that the continuous and discrete trajectories agree up to order $\alpha$.
Let us consider the function $\theta(t)$ and inspect its Taylor expansion starting from $\theta_0=\theta(0)$, for a jump $\Delta t=\alpha$ (1 iteration in discrete sense):
\begin{align*}
\theta(t) &\approx \theta_0 + \alpha \frac{\de \theta}{\de t}  + \frac{1}{2}\alpha^2 \frac{\de^2 \theta}{\de^2 t} \\
&= \theta_0 - \alpha \nabla\ell(\theta) + \alpha^2 g(\theta) + \frac{1}{2}\alpha^2 \frac{\de}{\de t} \bigg(-\nabla\ell(\theta) +{\color{magenta}\alpha g(\theta)}\bigg)\\
&\approx \theta_0 - \alpha \nabla\ell(\theta) + \alpha^2 \bigg(g(\theta) + \frac{1}{2}\bigg(-\nabla^2\ell(\theta)\frac{\de\theta}{\de t}\bigg)\bigg)\\
& = \theta_0 - \alpha \nabla\ell(\theta) + \alpha^2 \bigg(g(\theta) + \frac{1}{2}\bigg(\nabla^2\ell(\theta)\bigg(\nabla\ell(\theta) + {\color{magenta} \alpha g(\theta)}\bigg)\bigg)\bigg)
\end{align*}
where the terms in pink are of 3rd order in $\alpha$, hence disregarded.
Then, we impose $\theta(t) = \theta_{\tau+1}$, obtaining
\begin{align*}
    g(\theta) = -\frac{1}{2}\nabla^2\ell(\theta) \nabla\ell(\theta).
\end{align*}
Hence, the following continuous-time update rule agrees with the discrete-time update up to order $\alpha$
\begin{equation}
    \frac{\de \theta}{\de t} = -\nabla\ell(\theta) - \frac{1}{2}\alpha\nabla^2\ell(\theta)\nabla\ell(\theta).
\end{equation}
Lastly, we notice that this is the same as applying the usual continuous-time update rule to the modified loss
\begin{equation}
\tilde\ell(\theta) = \ell(\theta) + \frac{1}{4}\alpha\nabla\ell{(\theta)}^t \nabla\ell(\theta) = \ell(\theta) + \frac{1}{4}\alpha\big\Vert\nabla\ell(\theta) \big\Vert^2
\end{equation}
